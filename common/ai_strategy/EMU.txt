
EMU_save_the_rebellion = { # In the event that the rebellion isn't doing good
    enable = {
	    tag = EMU
		has_government = animalist
		is_in_faction = no
		has_war = no
		
		country_exists = EME
		AST = { has_war_with = EME }
		EME = {
		    has_government = animalist
			surrender_progress > 0.10
		}
		has_opinion = {
		    target = EME
			value > 0
		}
	}
	abort = {
		OR = {
			has_war_with = EME
			NOT = { country_exists = EME }
			is_in_faction_with = AST
			AST = {	
				NOT = { has_war_with = EME } 
			}
		}
	}
	
	ai_strategy = {
		type = contain
		id = "AST"			
		value = 200
	}
	
	ai_strategy = {
	    type = invade
		id = "AST"
		value = 500
	}
	
	ai_strategy = {
		type = protect
		id = "EME"			
		value = 200
	}
	ai_strategy = {
		type = support
		id = "EME"			
		value = 200
	}
}

    enable = {

		    has_war_with = AST
		}
	}

		    has_war_with = EME
			surrender_progress > 0.25
		}
	}

		id = "EME"
		value = 800
	}
}

EMU_save_the_aussies = { # In the event that the Emus ally with Australia
    enable = {
	    tag = EMU
		has_government = democratic
		is_in_faction = no
		has_war = no
		
		any_country = {
		    has_war_with = AST
		}
		AST = {
			has_government = democratic			
			surrender_progress > 0.15
		}
		has_opinion = {
			target = AST
			value > 0 # not if we hate AST
		}
	}
	abort = {
		OR = {
			has_war_with = AST
			any_country = {
			    is_in_faction_with = EMU
			}
			any_country = {
			    NOT = { has_war_with = AST }
			}
		}
	}
	
	ai_strategy = {
		type = contain
		id = "ROOT"			
		value = 200
	}
	
	ai_strategy = {
		type = protect
		id = "AST"			
		value = 200
	}
	ai_strategy = {
		type = support
		id = "AST"			
		value = 100
	}
}

EMU_support_animalists = { # Gotta support like-minded animals
    enable = {
	    tag = EMU
		has_government = animalist
		
		any_other_country = {
		    ROOT = {
			    has_government = animalist
			    has_war = yes    
			}
		}
	}
	abort = {
	    NOT = { has_government = animalist }
	}
	
	ai_strategy = {
	    type = support
		id = "ROOT"
		value = 2000
	}
	
	ai_strategy = {
	    type = protect
		id = "ROOT"
		value = 200
	}
}
